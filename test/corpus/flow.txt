MAIN
    DEFINE exit_code INTEGER
    DEFINE l_status INTEGER

    WHENEVER ANY ERROR GOTO _error
    GOTO _noerror

LABEL _error:
    LET l_status = STATUS
    LET exit_code = -1
    GOTO _exit

LABEL _noerror:
    LET exit_code = 0
    GOTO _exit

LABEL _exit:
    --EXIT PROGRAM exit_code

END MAIN
